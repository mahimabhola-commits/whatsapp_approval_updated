// WA Template Studio — Cloudflare Worker Proxy // Sits between GitHub Pages (browser) and Anthropic API. // Adds CORS headers so the browser can make the call. const CORS_HEADERS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, x-api-key, anthropic-version', }; export default { async fetch(request) { // CORS preflight if (request.method === 'OPTIONS') { return new Response(null, { headers: CORS_HEADERS }); } if (request.method !== 'POST') { return new Response('Method not allowed', { status: 405 }); } const apiKey = request.headers.get('x-api-key') || ''; if (!apiKey.startsWith('sk-ant-')) { return json({ error: { message: 'Missing or invalid Claude API key' } }, 401); } try { const body = await request.text(); const upstream = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': apiKey, 'anthropic-version': '2023-06-01', }, body, }); const data = await upstream.text(); return new Response(data, { status: upstream.status, headers: { 'Content-Type': 'application/json', ...CORS_HEADERS }, }); } catch (err) { return json({ error: { message: 'Proxy error: ' + err.message } }, 500); } } }; function json(obj, status) { return new Response(JSON.stringify(obj), { status, headers: { 'Content-Type': 'application/json', ...CORS_HEADERS }, }); }
